```python
# TOOLS/db/db_utils.py
"""共享 DB helper — 所有 TOOLS pipeline 脚本统一导入，消除复制粘贴。"""

import sqlite3
import sys
from pathlib import Path

# db/ → TOOLS/ → SRC/
_TOOLS_DIR = Path(__file__).resolve().parent.parent
_PROJECT_ROOT = _TOOLS_DIR.parent
DBS_DIR = _PROJECT_ROOT / "dbs"


def find_db(target: str) -> Path:
    """返回最新修改的匹配 DB 路径，找不到则 sys.exit。"""
    dbs = sorted(DBS_DIR.glob(f"{target}*.db"), key=lambda p: p.stat().st_mtime, reverse=True)
    if not dbs:
        sys.exit(f"[error] 找不到目标 DB: dbs/{target}*.db")
    return dbs[0]


def connect(db_path: "Path | str") -> sqlite3.Connection:
    """打开 SQLite 连接，启用 WAL + busy_timeout + row_factory。"""
    conn = sqlite3.connect(str(db_path))
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA busy_timeout=5000")
    conn.row_factory = sqlite3.Row
    return conn
```

```python
def mode_methods(url: str, conn: sqlite3.Connection, proxy: str | None, cookie_header: str | None = None) -> int:
    """测试非标准 HTTP 方法。

    OPTIONS: 仅当 Allow/ACAO 头中暴露 PUT/DELETE/PATCH/TRACE 时写 SP。
    PUT/DELETE/PATCH/TRACE: 仅当响应 2xx 时写 SP。
    HEAD: 永不写 SP（200 是正常行为）。
    """
    import urllib.request
    import urllib.error

    added = 0

    for method in HTTP_METHODS:
        try:
            req = urllib.request.Request(url, method=method)
            req.add_header("User-Agent", "Mozilla/5.0")
            if cookie_header:
                req.add_header("Cookie", cookie_header)

            if proxy:
                import urllib.request as ur
                handler = ur.ProxyHandler({"http": proxy, "https": proxy})
                opener = ur.build_opener(handler)
                resp = opener.open(req, timeout=10)
            else:
                resp = urllib.request.urlopen(req, timeout=10)

            code = resp.getcode()
            headers = dict(resp.headers)
            body = resp.read()
            size = len(body)

            if method == "OPTIONS":
                # 只关心 Allow / ACAO 是否暴露危险方法
                allow_raw = headers.get("Allow", "") + "," + headers.get("Access-Control-Allow-Methods", "")
                exposed_dangerous = {
                    m.strip().upper() for m in allow_raw.split(",")
                } & {"PUT", "DELETE", "PATCH", "TRACE"}
                if not exposed_dangerous:
                    continue  # OPTIONS 200 without dangerous methods is normal
                evidence = f"OPTIONS → HTTP {code}, Allow 暴露危险方法: {sorted(exposed_dangerous)}"
                reasoning = f"OPTIONS 响应头声明支持 {sorted(exposed_dangerous)}，可能暴露未授权写操作面"

            elif method in ("PUT", "DELETE", "PATCH", "TRACE"):
                if code not in (200, 201, 204):
                    continue
                evidence = f"{method} → HTTP {code}, body_len={size}"
                reasoning = f"{method} 返回 {code}，可能暴露未授权操作面"
            else:
                continue

            sp_id = new_sp_id("SP-PR")
            conn.execute(
                """INSERT INTO suspicious_points
                   (id, url, param, method, test_type, evidence, source, reasoning, risk, test_status, created_at)
                   VALUES (?, ?, ?, ?, 'method_tampering', ?, 'probe_runner', ?, 'Medium', 'untested', ?)
                   ON CONFLICT(id) DO NOTHING""",
                (sp_id, url, method, method, evidence, reasoning,
                 __import__("datetime").datetime.now().strftime("%Y-%m-%d %H:%M:%S")),
            )
            conn.commit()
            print(f"  [+] {method} {url} → {code} ({sp_id})")
            added += 1

        except Exception as e:
            e_str = str(e)
            if any(x in e_str for x in ("403", "404", "405", "501")):
                pass  # 正常拒绝
            else:
                print(f"  [-] {method} {url} → {e}")

    return added
```

```python
# TOOLS/db/cookie_helper.py
"""从 auth_sessions 表读取活跃 cookies，用于 pipeline 工具的 Cookie 头注入。"""

import sqlite3
import sys
from datetime import datetime, timezone
from urllib.parse import urlparse


def _domain_matches(cookie_domain: str, request_host: str) -> bool:
    """检查 cookie domain 是否匹配请求 host。支持 .example.com 泛匹配。"""
    cd = cookie_domain.lstrip(".").lower()
    rh = request_host.lower()
    return rh == cd or rh.endswith("." + cd)


def _path_matches(cookie_path: str, request_path: str) -> bool:
    """检查 cookie path 是否是请求 path 的前缀。空 path 或 '/' 匹配所有。"""
    if not cookie_path or cookie_path == "/":
        return True
    # RFC 6265: cookie-path must be a prefix of request-path
    if not request_path.startswith(cookie_path):
        return False
    # 避免 /api 匹配 /api2：前缀后必须是 '/' 或完全相等
    return len(request_path) == len(cookie_path) or request_path[len(cookie_path)] == "/"


def _is_expired(expires_at: str | None) -> bool:
    """returns True if cookie is expired. None/empty means session cookie (never expires)."""
    if not expires_at:
        return False
    try:
        # expires_at 可能是 ISO 格式字符串或 Unix 时间戳字符串
        ts = float(expires_at)
        # Playwright 用 -1 表示 session cookie
        if ts < 0:
            return False
        return ts < datetime.now(tz=timezone.utc).timestamp()
    except ValueError:
        pass
    try:
        # ISO 8601 格式
        exp = datetime.fromisoformat(expires_at.replace("Z", "+00:00"))
        return exp < datetime.now(tz=timezone.utc)
    except ValueError:
        return False  # 解析失败，保守处理：不过滤


def get_auth_cookies_dict(db_path: str, domain: str, request_path: str = "/") -> dict[str, str]:
    """返回匹配 domain+path 且未过期的活跃 cookies {name: value}，无匹配返回空 dict。"""
    try:
        conn = sqlite3.connect(db_path)
        conn.row_factory = sqlite3.Row
        rows = conn.execute(
            "SELECT token_name, token_value, domain, path, expires_at "
            "FROM auth_sessions WHERE is_active=1"
        ).fetchall()
        conn.close()
    except Exception as e:  # noqa: BLE001
        print(f"[cookie_helper] DB error ({db_path}): {e}", file=sys.stderr)
        return {}

    host = urlparse(domain if "://" in domain else f"http://{domain}").hostname or domain.split(":")[0]

    result = {}
    for row in rows:
        if row["domain"] and not _domain_matches(row["domain"], host):
            continue
        if not _path_matches(row["path"] or "/", request_path):
            continue
        if _is_expired(row["expires_at"]):
            continue
        result[row["token_name"]] = row["token_value"]
    return result


def get_auth_cookie_header(db_path: str, domain: str, request_path: str = "/") -> str | None:
    """返回 'name1=val1; name2=val2' 格式字符串，无匹配返回 None。"""
    d = get_auth_cookies_dict(db_path, domain, request_path)
    if not d:
        return None
    return "; ".join(f"{k}={v}" for k, v in d.items())
```
